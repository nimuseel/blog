import Image from 'next/image';
import {Button, Col} from '@nextui-org/react';
import {ProjectInfo} from '../../components/Career/ProjectInfo';
import {ProjectDetails} from '../../components/Career/ProjectDetails';
import {careerList} from "../../components/Career/data";
import {useRouter} from "next/router";

export const Zent = () => {
  const zentCareer = careerList.find((career) => career.companySlug === 'zent');
  const router = useRouter();

  return (
    <>
      <Button
        flat
        auto
        style={{backgroundColor: '#E5E5E5', marginBottom: '24px'}}
        onClick={router.back}>
        <Image src="/assets/icon/left-arrow.svg" alt="left arrow" width={16} height={16}/>
      </Button>
      <div style={{display: 'flex', flexDirection: 'column', gap: 12}}>
        {zentCareer.projects.map((project) => (
          <Col key={project.name}>
            <ProjectInfo project={project}/>
            <ProjectDetails project={project}/>
          </Col>
        ))}
      </div>
    </>
  );

};

<Zent/>
